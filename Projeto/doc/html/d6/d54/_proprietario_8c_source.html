307 int validaCPF(char *cpf)
308 {
309  // cpfs invalidos
310  char *cpfInval[] = {"000.000.000-00","111.111.111-11",
311  "222.222.222-22",
312  "333.333.333-33",
313  "444.444.444-44",
314  "555.555.555-55",
315  "666.666.666-66",

317  "888.888.888-88",
318  "999.999.999-99"};
319 
320  int cont,retorno = CPF_VALIDO,aux2,dig[11],soma=0,digVeri[2], contAux = 0;
321 
322  if(strlen(cpf) != TAM_CPF-1)
323  return CPF_INVALIDO;
324 
325 
326  // verifica se cpf so contem nros iguais
327  for(cont=0;cont<10;cont++)
328  {
329  if(strcmp(cpfInval[cont],cpf)==0)
330  {
331  retorno = CPF_INVALIDO;
332  break;
333  }
334  }
335 
336  if(retorno == CPF_INVALIDO)
337  return retorno;
338 
339  // transforma os caracteres do cpf em numeros
340  for(cont=0;cont<strlen(cpf);cont++)
341  {
342  if(cont != 3 && cont != 7 && cont != 11){
343  dig[contAux] = cpf[cont] - '0';
344  contAux++;
345  }
346  }
347 
348  // obtem o primeiro digito verificador
349  aux2 = 10;
350  for(cont=0;cont<9;cont++,aux2--)
351  {
352  soma += dig[cont] * aux2;
353  }
354  digVeri[0]=soma % 11;
355  digVeri[0]= digVeri[0]<2 ? 0 : 11-digVeri[0];
356 
357  // obtem o segundo digito verificador
358  soma=0;
359  aux2=11;
360  for(cont=0;cont<10;cont++,aux2--)
361  {
362  if(cont==9)
363  soma+= digVeri[0] * aux2;
364  else
365  soma+= dig[cont] * aux2;
366  }
367  digVeri[1]=soma % 11;
368  digVeri[1]= digVeri[1]<2 ? 0 : 11-digVeri[1];
369  if(digVeri[0]!=dig[9] || digVeri[1]!=dig[10])
370  {
371  retorno = CPF_INVALIDO;
372  }
